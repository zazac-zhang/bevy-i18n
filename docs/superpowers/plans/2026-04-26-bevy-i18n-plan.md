// In src/interpolate.rs — above the tests
use std::borrow::Cow;

/// Replace `{key}` placeholders in `template` with values from `vars`.
/// Unknown keys are left as-is (e.g. `{name}` stays `{name}`).
pub fn interpolate(template: &str, vars: &[(&str, &str)]) -> Cow<'_, str> {
    if vars.is_empty() || !template.contains('{') {
        return Cow::Borrowed(template);
    }

    let mut result = String::with_capacity(template.len());
    let mut chars = template.chars().peekable();

    while let Some(ch) = chars.next() {
        if ch == '{' {
            let mut key = String::new();
            let mut found_close = false;
            while let Some(&next) = chars.peek() {
                chars.next();
                if next == '}' {
                    found_close = true;
                    break;
                }
                key.push(next);
            }
            if found_close {
                if let Some((_, val)) = vars.iter().find(|(k, _)| *k == key) {
                    result.push_str(val);
                } else {
                    result.push('{');
                    result.push_str(&key);
                    result.push('}');
                }
            } else {
                result.push('{');
                result.push_str(&key);
            }
        } else {
            result.push(ch);
        }
    }

    Cow::Owned(result)
}

use std::collections::HashMap;
use std::io::{Cursor, Read};

use bevy::asset::{self, AssetLoader, AsyncReadExt};

/// A single locale's translation data, loaded as a Bevy Asset.
#[derive(Debug, Asset, Clone)]
pub struct I18nAsset {
    /// Flat key → translation string map.
    /// Keys use dot-notation: "game.title", "player.greeting"
    entries: HashMap<String, String>,
}

impl I18nAsset {
    /// Look up a translation by key.
    pub fn get(&self, key: &str) -> Option<&str> {
        self.entries.get(key).map(|s| s.as_str())
    }
}

/// Settings for the i18n asset loader (empty — no config needed).
#[derive(Debug, Default, Clone)]
pub struct I18nLoaderSettings;

/// AssetLoader that parses translation files into I18nAsset.
/// Supports YAML (default) and .po (feature: po).
#[derive(Default)]
pub struct I18nLoader;

#[cfg(feature = "yaml")]
impl AssetLoader for I18nLoader {
    type Asset = I18nAsset;
    type Settings = I18nLoaderSettings;
    type Error = Box<dyn std::error::Error + Send + Sync>;

    async fn load(
        &self,
        reader: &mut dyn Read,
        _settings: &I18nLoaderSettings,
        _load_context: &mut asset::LoadContext<'_>,
    ) -> Result<I18nAsset, Self::Error> {
        let mut buf = Vec::new();
        reader.read_to_end(&mut buf)?;
        let content = String::from_utf8_lossy(&buf);

        let yaml_value: serde_yaml::Value =
            serde_yaml::from_str(&content).map_err(|e| format!("YAML parse error: {e}"))?;

        let entries = flatten_yaml_value(&yaml_value);
        Ok(I18nAsset { entries })
    }

    fn extensions(&self) -> &[&str] {
        &["yaml", "yml"]
    }
}

#[cfg(feature = "po")]
impl AssetLoader for I18nLoader {
    type Asset = I18nAsset;
    type Settings = I18nLoaderSettings;
    type Error = Box<dyn std::error::Error + Send + Sync>;

    async fn load(
        &self,
        reader: &mut dyn Read,
        _settings: &I18nLoaderSettings,
        _load_context: &mut asset::LoadContext<'_>,
    ) -> Result<I18nAsset, Self::Error> {
        let mut buf = Vec::new();
        reader.read_to_end(&mut buf)?;
        let content = String::from_utf8_lossy(&buf);
        let entries = parse_po(&content);
        Ok(I18nAsset { entries })
    }

    fn extensions(&self) -> &[&str] {
        &["po"]
    }
}

/// Recursively flatten a serde_yaml::Value into dot-notation key-value pairs.
fn flatten_yaml_value(value: &serde_yaml::Value) -> HashMap<String, String> {
    let mut map = HashMap::new();
    flatten_recursive(value, String::new(), &mut map);
    map
}

fn flatten_recursive(
    value: &serde_yaml::Value,
    prefix: String,
    map: &mut HashMap<String, String>,
) {
    match value {
        serde_yaml::Value::String(s) => {
            map.insert(prefix, s.clone());
        }
        serde_yaml::Value::Mapping(m) => {
            for (key, val) in m {
                let key_str = match key {
                    serde_yaml::Value::String(s) => s.clone(),
                    serde_yaml::Value::Number(n) => n.to_string(),
                    _ => format!("{key:?}"),
                };
                let new_prefix = if prefix.is_empty() {
                    key_str
                } else {
                    format!("{prefix}.{key_str}")
                };
                flatten_recursive(val, new_prefix, map);
            }
        }
        other => {
            map.insert(prefix, format!("{other}"));
        }
    }
}

/// Minimal .po file parser. Handles msgid/msgstr and msgid_plural/msgstr[n].
#[cfg(feature = "po")]
fn parse_po(content: &str) -> HashMap<String, String> {
    let mut entries = HashMap::new();
    let mut current_msgid = String::new();
    let mut current_plural_id = String::new();
    let mut in_plural = false;

    for line in content.lines() {
        let line = line.trim();

        if line.starts_with("msgid_plural") {
            // Extract plural ID (remove prefix and quotes)
            current_plural_id = extract_po_string(line).unwrap_or_default();
            in_plural = true;
        } else if line.starts_with("msgid") && !line.starts_with("msgid_plural") {
            current_msgid = extract_po_string(line).unwrap_or_default();
            in_plural = false;
        } else if line.starts_with("msgstr[") {
            // Plural form: msgstr[n] "translation"
            if let Some((idx, val)) = parse_msgstr_indexed(line) {
                let plural_key = format!("{current_msgid}.{idx}");
                entries.insert(plural_key, val);
            }
        } else if line.starts_with("msgstr") && !in_plural {
            if let Some(val) = extract_po_string(line) {
                entries.insert(current_msgid.clone(), val);
            }
        }
    }

    entries
}

#[cfg(feature = "po")]
fn extract_po_string(line: &str) -> Option<String> {
    // Extract content between first and last quote
    let start = line.find('"')? + 1;
    let end = line.rfind('"')?;
    if start > end {
        return None;
    }
    Some(line[start..end].to_string())
}

#[cfg(feature = "po")]
fn parse_msgstr_indexed(line: &str) -> Option<(usize, String)> {
    // Parse: msgstr[N] "value"
    let bracket_start = line.find('[')?;
    let bracket_end = line.find(']')?;
    let idx: usize = line[bracket_start + 1..bracket_end].parse().ok()?;
    let val = extract_po_string(line)?;
    Some((idx, val))
}

```rust
// In src/resource.rs — above the tests
use std::collections::HashMap;
use std::sync::Arc;

use bevy::asset::{AssetId, Assets, Handle};

use crate::asset::I18nAsset;
use crate::interpolate::interpolate;

/// Central i18n resource — manages current locale, registered locale handles,
/// and a parsed cache for fast key lookup.
#[derive(Resource, Default)]
pub struct I18n {
    current_locale: String,
    locale_map: HashMap<String, Handle<I18nAsset>>,
    /// Cached parsed assets, populated lazily from Assets<I18nAsset>
    parsed_cache: HashMap<String, Arc<I18nAsset>>,
    /// Tracks whether locale changed since last frame
    locale_changed: bool,
}

impl I18n {
    /// Register a locale with its asset handle.
    pub fn add_locale(&mut self, locale: &str, handle: Handle<I18nAsset>) {
        self.locale_map
            .insert(locale.to_string(), handle.clone());
        self.parsed_cache.remove(locale);
    }

    /// Set the current locale. Returns true if changed.
    pub fn set_locale(&mut self, locale: &str) {
        if self.current_locale != locale {
            self.current_locale = locale.to_string();
            self.locale_changed = true;
        }
    }

    /// Get current locale identifier.
    pub fn current_locale(&self) -> &str {
        &self.current_locale
    }

    /// Check if locale has changed since last call.
    /// Call once per frame in the text update system.
    pub fn update_from(&mut self, locales: &Assets<I18nAsset>) -> bool {
        let was_changed = self.locale_changed;
        self.locale_changed = false;

        // Refresh parsed cache for the current locale if needed
        if self.locale_map.contains_key(&self.current_locale) {
            let handle = &self.locale_map[&self.current_locale];
            let asset_id = handle.id();
            // Only refresh if not already cached or cache is stale
            if !self.parsed_cache.contains_key(&self.current_locale) {
                if let Some(asset) = locales.get(asset_id) {
                    self.parsed_cache
                        .insert(self.current_locale.clone(), Arc::new(asset.clone()));
                }
            }
        }

        was_changed
    }

    /// Look up a translation key with optional variable interpolation.
    /// Returns the translated string, or the key itself if not found.
    pub fn get(
        &self,
        key: &str,
        vars: &[(&str, &str)],
        locales: &Assets<I18nAsset>,
    ) -> String {
        let asset = match self.get_parsed(&self.current_locale, locales) {
            Some(a) => a,
            None => return key.to_string(),
        };

        match asset.get(key) {
            Some(template) => interpolate(template, vars).into_owned(),
            None => key.to_string(), // fallback: return the key itself
        }
    }

    /// Get parsed locale from cache or assets.
    fn get_parsed<'a>(
        &'a self,
        locale: &str,
        locales: &'a Assets<I18nAsset>,
    ) -> Option<&'a Arc<I18nAsset>> {
        if let Some(cached) = self.parsed_cache.get(locale) {
            return Some(cached);
        }
        // Check if the handle exists but the asset isn't loaded yet
        self.locale_map.get(locale).and_then(|handle| {
            locales.get(handle.id()).map(|asset| {
                // We'd insert into cache here, but can't mutate self
                // The update_from handles caching; this is a fallback
                static_fallback(asset)
            })
        })
    }
}

/// Fallback for uncached lookups — returns a static reference.
/// In practice, update_from should populate the cache first.
fn static_fallback(asset: &I18nAsset) -> &'static Arc<I18nAsset> {
    // This is a simplified approach; in production you'd use
    // a different pattern. For now, the cache handles it.
    // This function exists for type compatibility only.
    leak_arc(asset)
}

fn leak_arc(asset: &I18nAsset) -> &'static Arc<I18nAsset> {
    let arc = Arc::new(asset.clone());
    // Safety: we leak intentionally here to get a 'static reference.
    // This is acceptable because locale assets are long-lived.
    Box::leak(Box::new(arc))
}

use std::collections::HashMap;

use bevy::asset::{Assets, Handle};

use crate::asset::I18nAsset;
use crate::interpolate::interpolate;

/// Central i18n resource — manages current locale and registered locale handles.
#[derive(Resource, Default)]
pub struct I18n {
    current_locale: String,
    locale_map: HashMap<String, Handle<I18nAsset>>,
    locale_changed: bool,
}

impl I18n {
    /// Register a locale with its asset handle.
    pub fn add_locale(&mut self, locale: &str, handle: Handle<I18nAsset>) {
        self.locale_map.insert(locale.to_string(), handle);
    }

    /// Set the current locale.
    pub fn set_locale(&mut self, locale: &str) {
        if self.current_locale != locale {
            self.current_locale = locale.to_string();
            self.locale_changed = true;
        }
    }

    /// Get current locale identifier.
    pub fn current_locale(&self) -> &str {
        &self.current_locale
    }

    /// Check if locale has changed since last call.
    /// Call once per frame in the text update system.
    pub fn update_from(&mut self) -> bool {
        let was_changed = self.locale_changed;
        self.locale_changed = false;
        was_changed
    }

    /// Look up a translation key with optional variable interpolation.
    /// Returns the translated string, or the key itself if not found.
    pub fn get(&self, key: &str, vars: &[(&str, &str)], locales: &Assets<I18nAsset>) -> String {
        let Some(handle) = self.locale_map.get(&self.current_locale) else {
            return key.to_string();
        };

        let Some(asset) = locales.get(handle.id()) else {
            return key.to_string();
        };

        let Some(template) = asset.get(key) else {
            return key.to_string();
        };

        interpolate(template, vars).into_owned()
    }

    /// Check if a locale's asset is loaded.
    pub fn is_locale_loaded(&self, locale: &str, locales: &Assets<I18nAsset>) -> bool {
        self.locale_map
            .get(locale)
            .and_then(|h| locales.get(h.id()))
            .is_some()
    }
}
```

This is much cleaner. No cache, no Arc, no leak. The `Assets<I18nAsset>` lookup by Handle ID is O(1) and the `HashMap` in `I18nAsset` lookup is also O(1). For game UI text (updated maybe 60fps with a few dozen entities), this is plenty fast.

// In src/component.rs — above the tests
use bevy::prelude::Component;

/// Component that marks a Text entity for automatic translation.
///
/// Add this alongside Bevy's `Text` component to have the text
/// content automatically resolved from the current locale.
///
/// # Example
/// ```ignore
/// commands.spawn((
///     Text::new(""),
///     T::new("game.title"),
/// ));
/// ```
#[derive(Component, Clone, Debug)]
pub struct T {
    /// Translation key (e.g. "game.title")
    pub key: String,
    /// Variable substitutions (key → value)
    pub vars: Vec<(String, String)>,
    /// Whether the text needs to be re-resolved
    pub dirty: bool,
}

impl T {
    /// Create a T component for a static translation key.
    pub fn new(key: impl Into<String>) -> Self {
        Self {
            key: key.into(),
            vars: Vec::new(),
            dirty: true,
        }
    }

    /// Create a T component with variable substitutions.
    ///
    /// # Example
    /// ```ignore
    /// T::with_vars("player.greeting", &[("name", "Hero")])
    /// ```
    pub fn with_vars(key: impl Into<String>, vars: &[(&str, &str)]) -> Self {
        Self {
            key: key.into(),
            vars: vars
                .iter()
                .map(|(k, v)| (k.to_string(), v.to_string()))
                .collect(),
            dirty: true,
        }
    }

    /// Mark this component as needing re-resolution.
    pub fn mark_dirty(&mut self) {
        self.dirty = true;
    }
}

// In src/systems.rs
use bevy::asset::Assets;
use bevy::prelude::*;
use bevy::text::Text;

use crate::asset::I18nAsset;
use crate::component::T;
use crate::resource::I18n;

/// Resolves translation keys for all T components.
/// Marks dirty T components that need text updates.
pub fn resolve_translations(
    mut i18n: ResMut<I18n>,
    locales: Res<Assets<I18nAsset>>,
    mut query: Query<&mut T>,
) {
    let locale_changed = i18n.update_from();

    if locale_changed {
        for mut t in &mut query {
            t.mark_dirty();
        }
    }
}

/// Updates Text content for dirty T components.
pub fn update_text_system(
    i18n: Res<I18n>,
    locales: Res<Assets<I18nAsset>>,
    mut query: Query<(&T, &mut Text)>,
) {
    for (t, mut text) in &mut query {
        if !t.dirty {
            continue;
        }

        let vars: Vec<(&str, &str)> =
            t.vars.iter().map(|(k, v)| (k.as_str(), v.as_str())).collect();

        let translated = i18n.get(&t.key, &vars, &locales);

        // Clear existing sections and add the translated text
        text.0.clear();
        text.0.push(bevy::text::TextSection::new(
            translated,
            text.0.first().map(|s| s.style.clone()).unwrap_or_default(),
        ));
    }
}
